Condensed Datepicker

See code
HTML/JS

To create a condensed datepicker, just add the condensed modifier to the datepicker block and the control element.

<div class="l-container">
  <div class="form">
    <div class="form__group">
      <label class="form__group__label">
        <div class="form__group__label__text">Form label</div>
      </label>
      <div class="form__group__control">
        <div id="myCondensedDatepicker" data-datepicker-watch="#myCondensedDatepickerInput" data-datepicker-display-week="data-datepicker-display-week" data-datepicker-icon-prev="/images/icons.svg#arrow-left" data-datepicker-icon-next="/images/icons.svg#arrow-right" class="datepicker datepicker--condensed">
          <div data-datepicker="#myCondensedDatepicker" class="datepicker__trigger">
            <svg class="datepicker__trigger__icon">
              <use xlink:href="/images/icons.svg#calendar"></use>
            </svg>
          </div>
          <div class="datepicker__input">
            <input type="text" id="myCondensedDatepickerInput" class="control control--input control--condensed"/>
          </div>
        </div>
      </div>
    </div>
  </div>
</div>
